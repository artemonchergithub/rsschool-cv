Task: Greed is Good

 

public class Greed {
  public static int greedy(int[] dice){
    int sum = 0;
    int[] repeats = new int[6];
    for (int die : dice) {
        repeats[die - 1]++;
    }
    sum += repeats[0] / 3 * 1000 + repeats[0] % 3 * 100;
    for (int i = 1; i < 6; i++) {
        sum += repeats[i] / 3 * (i + 1) * 100;
    }
    sum += repeats[4] % 3 * 50;
    return sum;
  }
}
